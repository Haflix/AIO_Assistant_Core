- check if plugin exists in given path
- check if all things are given (especially asynced etc)

config.yml:::
plugins:
  - name: str
    enabled: boolean
    path: str | Empty
    version: str | Empty so it doesnt check for version
    arguments: 
        [Any] | Empty

  - name: str
    enabled: boolean
    path: str | Empty
    version: str | Empty
    arguments: 
        [Any] | Empty

mqtt:
    enabled: boolean
    hostname: str | Empty for standard
    broker_ip: str
    port: int

general:
    ident_name: str | Empty for same as hostname
    log_path: str | Empty to disable
    plugin_package: str
